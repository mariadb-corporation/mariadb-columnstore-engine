select l_extendedprice, l_discount
from   lineitem
where  l_shipdate >= to_date('1993-01-01','yyyy-mm-dd')
       and l_shipdate < to_date('1994-01-01','yyyy-mm-dd') 
       and l_discount between 0.05 and  0.07
       and l_quantity < 24;
